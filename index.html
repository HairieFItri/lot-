// --- REPAIR: HIGHLIGHT LOT BAGI NAMPAK PETAK (KUNING TERANG) --- async function generateFullPDF() { const { jsPDF } = window.jspdf; const noLot = document.getElementById('val-lot').innerText; const pdfBtn = document.getElementById('pdf-btn'); if(noLot === "-" || !selectedLotData) { alert("Sila pilih satu lot di dalam peta terlebih dahulu!"); return; } // Disable butang & tunjuk loading pdfBtn.disabled = true; pdfBtn.innerHTML = ' Sedang Proses...'; // STEP 1: Sembunyikan semua layer kecuali Satelit map.removeLayer(lotLayer); if(map.hasLayer(boundaryPoints)) map.removeLayer(boundaryPoints); // ========================================================= // STEP 2: Ubah suai Highlight Petak Lot (DIBERSIHKAN & DITERANGKAN) // Kita guna Kuning Terang untuk garisan, dan kosongkan isi // ========================================================= var singleLotLayer = L.geoJson(selectedLotData, { style: { color: '#FFFF00', // Garisan Kuning Terang (Pure Yellow) weight: 5, // Garisan tebal sikit bagi jelas (asal 4) opacity: 1, // Garisan penuh (tak lutsinar) fillColor: 'none', // TIADA WARNA ISI (supaya bumbung nampak) fillOpacity: 0 // Pastikan lutsinar 100% untuk isi } }).addTo(map); try { // Beri masa sekejap untuk map 'settle' await new Promise(r => setTimeout(r, 600)); const canvas = await html2canvas(document.getElementById('map'), { useCORS: true, scale: 2, // Kualiti HD ignoreElements: (node) => node.classList.contains('leaflet-control-zoom') }); const imgData = canvas.toDataURL('image/png'); const doc = new jsPDF('p', 'mm', 'a4'); // Design Header PDF doc.setFillColor(183, 28, 28); doc.rect(0, 0, 210, 35, 'F'); doc.setTextColor(255, 255, 255); doc.setFontSize(20); doc.text("LAPORAN RASMI LOT TANAH", 105, 18, {align: 'center'}); doc.setFontSize(10); doc.text("Taman Putra Harmoni Resident | Sistem GIS", 105, 26, {align: 'center'}); // Masukkan Gambar Peta (Satelit + Petak Lot Kuning Terang) doc.setDrawColor(183, 28, 28); doc.setLineWidth(0.5); doc.rect(14.5, 44.5, 181, 101); // Border luar doc.addImage(imgData, 'PNG', 15, 45, 180, 100); // Kotak Info Maklumat doc.setFillColor(245, 245, 245); doc.rect(15, 155, 180, 45, 'F'); doc.setTextColor(0, 0, 0); doc.setFontSize(14); doc.setFont("helvetica", "bold"); doc.text("PERINCIAN MAKLUMAT:", 20, 165); doc.setFontSize(12); doc.setFont("helvetica", "normal"); doc.text("Nombor Lot : " + noLot, 20, 175); doc.text("Nama Pemilik : " + document.getElementById('val-owner').innerText, 20, 183); doc.text("Luas Kawasan : " + document.getElementById('val-area').innerText, 20, 191); // Footer doc.setFontSize(9); doc.setTextColor(150, 150, 150); doc.text("Laporan ini dijana secara digital pada: " + new Date().toLocaleString(), 105, 285, {align: 'center'}); doc.save("Laporan_Lot_" + noLot + ".pdf"); } catch (e) { console.error(e); alert("Ralat teknikal semasa export PDF."); } finally { // STEP 3: Kembalikan keadaan asal peta map.removeLayer(singleLotLayer); if(document.getElementById('check-lot').checked) map.addLayer(lotLayer); if(document.getElementById('check-boundary').checked) map.addLayer(boundaryPoints); pdfBtn.disabled = false; pdfBtn.innerHTML = ' Export PDF'; } }
